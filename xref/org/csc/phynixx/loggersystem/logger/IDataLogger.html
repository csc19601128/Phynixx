27  import java.io.IOException;

29  /**
30   * A logger ist able to write data in an atomic manner. The data is entirely written or the data ist rejected.
31   *
32   * This class is not thread safe . Use facades to protect instances
33   */
34  public interface IDataLogger {
35  
36      /**
37       * Sub-classes call this method to write log records with
38       * a specific record type.
39       *
40       * @param type a record type defined in LogRecordType.
41       * @param data record data to be logged.
42       * @return a log key that can be used to de-reference the record.
43       * <p/>
44       */
45      long write(short type, byte[][] data)
46              throws InterruptedException, IOException;
47  
48  
49      /**
50       * callback method to replay the data of the logger.
51       * @param replayListener
52       * @throws IOException
53       */
54      void replay(ILogRecordReplayListener replayListener) throws IOException;
55  
56      /**
57       * close the Log files and perform necessary cleanup tasks.
58       * The logger could be reopended
59       */
60      void close() throws IOException, InterruptedException;
61  
62      boolean isClosed();
63  
64      /**
65       * opens the logger with the specified ACCESS_MODE. If the logger isn't closed it is closed.
66       *
67       * @param accessMode
68       * @throws IOException
69       *
70       * @see #reopen(org.csc.phynixx.loggersystem.logger.channellogger.AccessMode)
71       */
72      void open(AccessMode accessMode) throws IOException;
73  
74      /**
75       * reopens the datalogger. It is assumed that the logger is open.
76       * <pre>
77       *    READ   - position to position 0
78       *    WRITE  - position to position 0 and resets the committed size to 0
79       *    APPEND - position to the committed size
80       * </pre>
81       *
82       * @param accessMode
83       * @throws IOException
84       * @throws InterruptedException
85       */
86      void reopen(AccessMode accessMode) throws IOException, InterruptedException;
87  
88  
89      /**
90       * removes the logger an all its resources
91       *
92       * @throws IOException
93       */
94      void destroy() throws IOException;
95  }
