38  import javax.net.ssl.*;
39  import java.io.*;

45  /**
46   * Class used to add the server's certificate to the KeyStore
47   * with your trusted certificates.
48   */
49  public class InstallCert {
50  
51      public static void main(String[] args) throws Exception {
52          String host;
53          int port;
54          char[] passphrase;
55          if ((args.length == 1) || (args.length == 2)) {
56              String[] c = args[0].split(":");
57              host = c[0];
58              port = (c.length == 1) ? 443 : Integer.parseInt(c[1]);
59              String p = (args.length == 1) ? "changeit" : args[1];
60              passphrase = p.toCharArray();
61          } else {
62              System.out.println("Usage: java InstallCert [:port] [passphrase]");
63              return;
64          }
65  
66          File file = new File("jssecacerts");
67          if (!file.isFile()) {
68              char SEP = File.separatorChar;
69              File dir = new File(System.getProperty("java.home") + SEP
70                      + "lib" + SEP + "security");
71              file = new File(dir, "jssecacerts");
72              if (!file.isFile()) {
73                  file = new File(dir, "cacerts");
74              }
75          }
76          System.out.println("Loading KeyStore " + file + "...");
77          InputStream in = new FileInputStream(file);
78          KeyStore ks = KeyStore.getInstance(KeyStore.getDefaultType());
79          ks.load(in, passphrase);
80          in.close();
81  
82          SSLContext context = SSLContext.getInstance("TLS");
83          TrustManagerFactory tmf =
84                  TrustManagerFactory.getInstance(TrustManagerFactory.getDefaultAlgorithm());
85          tmf.init(ks);
86          X509TrustManager defaultTrustManager = (X509TrustManager) tmf.getTrustManagers()[0];
87          SavingTrustManager tm = new SavingTrustManager(defaultTrustManager);
88          context.init(null, new TrustManager[]{tm}, null);
89          SSLSocketFactory factory = context.getSocketFactory();
90  
91          System.out.println("Opening connection to " + host + ":" + port + "...");
92          SSLSocket socket = (SSLSocket) factory.createSocket(host, port);
93          socket.setSoTimeout(10000);
94          try {
95              System.out.println("Starting SSL handshake...");
96              socket.startHandshake();
97              socket.close();
98              System.out.println();
99              System.out.println("No errors, certificate is already trusted");
100         } catch (SSLException e) {
101             System.out.println();
102             e.printStackTrace(System.out);
103         }
104 
105         X509Certificate[] chain = tm.chain;
106         if (chain == null) {
107             System.out.println("Could not obtain server certificate chain");
108             return;
109         }
110 
111         BufferedReader reader =
112                 new BufferedReader(new InputStreamReader(System.in));
113 
114         System.out.println();
115         System.out.println("Server sent " + chain.length + " certificate(s):");
116         System.out.println();
117         MessageDigest sha1 = MessageDigest.getInstance("SHA1");
118         MessageDigest md5 = MessageDigest.getInstance("MD5");
119         for (int i = 0; i < chain.length; i++) {
120             X509Certificate cert = chain[i];
121             System.out.println
122                     (" " + (i + 1) + " Subject " + cert.getSubjectDN());
123             System.out.println("   Issuer  " + cert.getIssuerDN());
124             sha1.update(cert.getEncoded());
125             System.out.println("   sha1    " + toHexString(sha1.digest()));
126             md5.update(cert.getEncoded());
127             System.out.println("   md5     " + toHexString(md5.digest()));
128             System.out.println();
129         }
130 
131         System.out.println("Enter certificate to add to trusted keystore or 'q' to quit: [1]");
132         String line = reader.readLine();
133         int k;
134         try {
135             k = (line !=null && line.length() == 0) ? 0 : Integer.parseInt(line.trim()) - 1;
136         } catch (NumberFormatException e) {
137             System.out.println("KeyStore not changed");
138             return;
139         }
140 
141         X509Certificate cert = chain[k];
142         String alias = host + "-" + (k + 1);
143         ks.setCertificateEntry(alias, cert);
144 
145         OutputStream out = new FileOutputStream("jssecacerts");
146         ks.store(out, passphrase);
147         out.close();
148 
149         System.out.println();
150         System.out.println(cert);
151         System.out.println();
152         System.out.println
153                 ("Added certificate to keystore 'jssecacerts' using alias '"
154                         + alias + "'");
155     }

157     private static final char[] HEXDIGITS = "0123456789abcdef".toCharArray();
158 
159     private static String toHexString(byte[] bytes) {
160         StringBuilder sb = new StringBuilder(bytes.length * 3);
161         for (int b : bytes) {
162             b &= 0xff;
163             sb.append(HEXDIGITS[b >> 4]);
164             sb.append(HEXDIGITS[b & 15]);
165             sb.append(' ');
166         }
167         return sb.toString();
168     }

170     private static class SavingTrustManager implements X509TrustManager {
171 
172         private final X509TrustManager tm;
173         private X509Certificate[] chain;
174 
175         SavingTrustManager(X509TrustManager tm) {
176             this.tm = tm;
177         }
178 
179         public X509Certificate[] getAcceptedIssuers() {
180             throw new UnsupportedOperationException();
181         }
182 
183         public void checkClientTrusted(X509Certificate[] chain, String authType)
184                 throws CertificateException {
185             throw new UnsupportedOperationException();
186         }
187 
188         public void checkServerTrusted(X509Certificate[] chain, String authType)
189                 throws CertificateException {
190             this.chain = chain;
191             tm.checkServerTrusted(chain, authType);
192         }
193     }
