179     /**
180      * Schliesst die Datei und den FileChannel
181      *
182      * @throws java.io.IOException , wenn das Schliessen schief geht.
183      */
184     public void close() throws IOException {
185         if (raf != null) {
186             // gibt Lock auf datei frei
187             try {
188                 if (this.fileLock != null) {
189                     if( fileLock.isValid()) {
190                      this.fileLock.release();
191                     } else {
192                         LOG.error("Filelock not valid");
193                     }
194                 } else {
195                     LOG.error("Kein Filelock gesetzt");
196                 }
197             } finally {
198                 // Schliessen der Daten-Datei
199                 this.fileLock = null;
200                 raf.close();
201                 raf = null;
202             }
203         }
204     }

279     /**
280      * kopiert den Bereich zwischen startPosition und enedPosition der Datei in den ByteBuffer.
281      * Es wird die Read-methode des RandomAccessFiles genommen, da sich
282      * herausgestellt hat, dass die Methode im Batchbetrieb etwas schneller
283      * ist als die Channel-methode.
284      * Ee werden bytes gelesen fuer die gilt: <br>
285      * startPosition <= b < endPosition
286      *
287      * @param length Anzahl der zu lesenden Zeichen
288      *               werden soll
289      * @return ByteBuffer mit dem Inhalt der Datei zwischen startPosition und enedPosition
290      * @throws java.io.IOException IO-Fehler
291      */
292     public byte[] read(long length) throws IOException {
293         check();
294 
295         if (length >= new Long(Integer.MAX_VALUE).longValue()) {
296             throw new IllegalArgumentException("Length of read area may not exeed " + Integer.MAX_VALUE);
297         }
298 
299         int intLength = new Long(length).intValue();
300 
301         byte[] buffer = new byte[intLength];
302 
303         if (isClose()) {
304             throw new IOException("TAEnabledChanneList geschlossen");
305         }
306         // ueberpruefung der Eingabeargumente
307         if (buffer.length < intLength) {
308             throw new IllegalArgumentException("Uebergebener Buffer zu klein fuer angeforderten Lesezugriff : " + intLength);
309         }
310 
311         long retVal = this.getRandomAccessFile().read(buffer, 0, intLength);
312 
313         if (retVal < 0) {
314             throw new IOException("Channel kann nicht bis zur Position " + (position() + intLength) + " lesen");
315         }
316         return buffer;
317     }

320      * Schreibt den Inhalt des uebergebenen ByteBuffers in die Datei. Achtung:
321      * ruft zuerst flip auf. Setzt nach dem Schreiben die Position auf 0 und
322      * das Limit auf die Kapazitaet des ByteBuffers (zur weiteren Verwendung)
323      *
324      * @param buffer , der in die Datei zu schreiben ist.
325      * @throws java.io.IOException , falls irgendetwas beim Schreiben schief geht
326      */
327     public void write(byte[] buffer) throws IOException {
328         check();
329 
330         //assert buffer != null : "Der Buffer ist null";
331         if (isClose()) {
332             throw new IOException("TAEnabledChanneList geschlossen");
333         }
334         long currentPosition = this.position();
335         this.raf.write(buffer);
336         // this.incPosition(buffer.length);
337 
338         assert this.position() - currentPosition == buffer.length : "Expected new position : " + currentPosition + buffer.length + " actual position " + this.position();
339

341     }

475     /**
476      * Stellt die Comittet Dateigroesse wieder her. Die Comittete Dateigroesse
477      * steht jeweils am Anfang der Datei. Positioniert auf die neu gesetzte
478      * fileSize.
479      *
480      * @throws java.io.IOException IO-Fehler
481      * @see
482      */
483     private void restoreCommittedSize() throws IOException {
484 
485         check();
486         long privCommittedSize=-1;
487         this.getRandomAccessFile().seek(0);
488         if (this.raf.length() < HEADER_LENGTH) {
489             this.getRandomAccessFile().writeLong(0);
490             privCommittedSize=0;
491         } else {
492             privCommittedSize = this.getRandomAccessFile().readLong();
493         }
494         this.position(privCommittedSize);
495     }
