32      public static void sleep(long msecs) {
33          long start = System.currentTimeMillis();
34          long waiting = msecs;
35          while (waiting > 0) {
36              try {
37                  Thread.currentThread().sleep(waiting);
38              } catch (InterruptedException e) {
39              } finally {
40                  waiting = msecs - (System.currentTimeMillis() - start);
41              }
42          }
43      }

49      public static void configureLogging(String log4jLevel) {
50          Set<String> validLogLevels = new HashSet<String>();
51          validLogLevels.add("DEBUG");
52          validLogLevels.add("INFO");
53          validLogLevels.add("WARNING");
54          validLogLevels.add("ERROR");
55          validLogLevels.add("FATAL");
56  
57          Properties log4jProps = new Properties();
58          InputStream io = Thread.currentThread().getContextClassLoader().getResourceAsStream("log4j.properties");
59          if (io == null) {
60              throw new IllegalStateException("log4j.properties not found in ClassPath for testing ");
61          }
62          try {
63              log4jProps.load(io);
64          } catch (IOException e) {
65              e.printStackTrace();
66          }
67  
68          // Reinitialize the logger and substitute any environment pros
69          Properties props = System.getProperties();
70          // copy all props beginning with log4j
71          Iterator<Map.Entry<Object,Object>> iter = props.entrySet().iterator();
72          while (iter.hasNext()) {
73              Map.Entry<Object,Object> entry = iter.next();
74              String propName = (String) entry.getKey();
75              if (propName.startsWith("log4j")) {
76                  log4jProps.put(entry.getKey(), entry.getValue());
77              }
78          }
79  
80          // check if log4j.level is set correctly ; else set the Default to INFO
81          String level = log4jProps.getProperty("log4j_level");
82          if (level == null || !validLogLevels.contains(level)) {
83              level = log4jLevel;
84              log4jProps.setProperty("log4j_level", level);
85          }
86  
87          org.apache.log4j.LogManager.resetConfiguration();
88  
89          PropertyConfigurator.configure(log4jProps);
90  
91      }
