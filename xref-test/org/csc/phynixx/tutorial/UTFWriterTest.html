75      /**
76       * messages ' My message {0}' are written concurrently. To check, that the message are written once and haven't been disrupted , read all messages,
77       * order them and check, that each mwssage apperras once.
78       *
79       * @throws Exception
80       */
81      @Test
82      public void testLocking() throws Exception {
83  
84          final File file = this.tmpDir.assertExitsFile("testOut.txt");
85          UTFWriter writer = new UTFWriterImpl(file);
86          final ExecutorService executorService = Executors.newFixedThreadPool(10);
87          Set<TestCallable> callables = new HashSet<TestCallable>();
88          for (int i = 0; i < 30; i++) {
89              callables.add(new TestCallable(i, writer));
90          }
91          final List<Future<String>> futures = executorService.invokeAll(callables);
92  
93          final List<String> content = writer.readContent();
94          Collections.sort(content);
95          for (int i = 0; i < content.size(); i++) {
96              String line = content.get(i);
97              Assert.assertEquals("My message " + format.format(i), line);
98          }
99      }
100 
101     private static class TestCallable implements Callable<String> {
102 
103         private final UTFWriter writer;
104 
105         private final int id;
106 
107         private TestCallable(int id, UTFWriter writer) {
108             this.writer = writer;
109             this.id = id;
110         }
111 
112         /**
113          * Computes a result, or throws an exception if unable to do so.
114          *
115          * @return computed result
116          * @throws Exception if unable to compute a result
117          */
118         @Override
119         public String call() throws Exception {
120             String lockToken = writer.lock();
121             try {
122                 writer.write("My message " + format.format(id));
123                 System.out.println("My message " + format.format(id));
124                 //Thread.sleep(1000);
125                 return lockToken;
126             } finally {
127                 writer.unlock(lockToken);
128             }
129         }
