60          File file = this.tmpDir.assertExitsFile("my_test.tmp");
61          UTFWriter utfWriter= new UTFWriterImpl(file);

63          TAEnabledUTFWriterImpl writer = new TAEnabledUTFWriterImpl("recover", utfWriter);
64          writer.resetContent();
65  
66          try {
67              writer.write("AA").write("BB");
68          } finally {
69              writer.close();
70          }
71  
72          TAEnabledUTFWriterImpl recoverWriter = new TAEnabledUTFWriterImpl("recover", utfWriter);
73          try {
74              List<String> content = recoverWriter.readContent();
75              Assert.assertEquals(2, content.size());
76              Assert.assertEquals("AA", content.get(0));
77              Assert.assertEquals("BB", content.get(1));

80              writer.close();
