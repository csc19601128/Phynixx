45      private final String MESSAGE2 = "QWERT";

53      public void setUp() throws Exception {
54          // configuring the log-system (e.g. log4j)
55          TestUtils.configureLogging();

57          // delete all tmp files ...
58          this.tmpDir = new TmpDirectory("howllogger");
59          this.tmpDir.clear();

62          System.getProperties().setProperty("howl.log.logFileDir", this.tmpDir.getDirectory().getCanonicalPath());
63  
64          this.loggerFactory = new FileChannelDataLoggerFactory("test", tmpDir.getDirectory());
65  
66      }

69      public void tearDown() throws Exception {
70          // delete all tmp files ...
71          this.tmpDir.clear();

74      @Test
75      public void testWriteData() throws Exception {
76  
77          IDataLogger dataLogger1 = this.loggerFactory.instanciateLogger("log");
78          PhynixxXADataRecorder dataRecorder = PhynixxXADataRecorder.openRecorderForWrite(1, new XADataLogger(dataLogger1), null);
79  
80          LogRecordWriter logRecordWriter2 = new LogRecordWriter();
81          logRecordWriter2.writeUTF(MESSAGE2);
82          byte[][] content2 = new byte[1][];
83          content2[0] = logRecordWriter2.toByteArray();
84          dataRecorder.writeRollbackData(content2);
85  
86          LogRecordWriter logRecordWriter1 = new LogRecordWriter();
87          logRecordWriter1.writeUTF(MESSAGE1);
88          byte[][] content1 = new byte[1][];
89          content1[0] = logRecordWriter1.toByteArray();
90          dataRecorder.writeRollforwardData(content1);
91  
92          dataRecorder.close();
93  
94          IDataLogger dataLogger2 = this.loggerFactory.instanciateLogger("log");
95          dataRecorder = PhynixxXADataRecorder.recoverDataRecorder(new XADataLogger(dataLogger2), null);
96          dataRecorder.replayRecords(new IDataRecordReplay() {
97              @Override
98              public void replayRollback(IDataRecord record) {
99                  LogRecordReader logReader = new LogRecordReader(record.getData()[0]);
100                 String data = null;
101                 try {
102                     data = logReader.readUTF();
103                 } catch (IOException e) {
104                     throw new DelegatedRuntimeException(e);
105                 }
106                 Assert.assertEquals(MESSAGE2, data);
107                 System.out.println(data);
108             }
109 
110             @Override
111             public void replayRollforward(IDataRecord record) {
112                 LogRecordReader logReader = new LogRecordReader(record.getData()[0]);
113                 String data = null;
114                 try {
115                     data = logReader.readUTF();
116                 } catch (IOException e) {
117                     throw new DelegatedRuntimeException(e);
118                 }
119                 Assert.assertEquals(MESSAGE1, data);
120                 System.out.println(data);
121             }
122         });
