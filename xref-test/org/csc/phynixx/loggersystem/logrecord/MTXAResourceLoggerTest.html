62          public void replayRollback(IDataRecord message) {
63              if (message.getOrdinal().intValue() == 1) {
64                  content = new String(message.getData()[0]);
65              } else {
66                  String part = new String(message.getData()[0]);
67                  contentParts.append(part);
68              }

72          }
