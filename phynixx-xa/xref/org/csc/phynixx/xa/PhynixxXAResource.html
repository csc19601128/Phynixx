95              };
96              // it is registered at the resource xaResourceFactory's watchdog  ....
97              xaResourceFactory.registerWatchCondition(this.timeoutCondition);
98          }

103     /**
104      * called when the current XAresource is expired (time out occurred)
105      * The current Impl. does nothing but marked the associated TX as rollback only
106      * The underlying core connection are not treated ....
107      * <p/>
108      * There are two different situation that have to be handled
109      * If the connection expires because of a long running method call the connection isn't
110      * treated and the XAResource is marked as rollback only. The rollback is done by the Transactionmanager.
111      * If the TX expires because the Transactionmanagers doesn't answer, the underlying connection is rolledback
112      * and the XAResource is marked as rollback only (N.B. rolling back the connection
113      * marks the XAResource as heuristically rolled back)
114      */
115     public void conditionViolated() {
116         try {
117             if (this.xaConnectionHandle != null) {
118                 XATransactionalBranch<C> transactionalBranch = this.xaConnectionHandle.toGlobalTransactionBranch();
119                 if (transactionalBranch == null) {
120                     transactionalBranch.setRollbackOnly(true);
121                     }
122             }
123             if (LOG.isInfoEnabled()) {
124                 String logString = "PhynixxXAResource.expired :: XAResource " + this.getId() + " is expired (time out occurred) and all associated TX are rollbacked ";
125                 LOG.info(new ConditionViolatedLog(this.timeoutCondition, logString).toString());
126             }
127 
128         } finally {
129             // no monitoring anymore
130             setTimeOutActive(false);
131         }
132 
133     }
134 
135     private void setTimeOutActive(boolean active) {
136         if(this.timeoutCondition!=null) {
137         this.setTimeOutActive(active);
138         }
139     }

165     /**
166      * transaction branch is interchangeably with transaction context
167      * <p/>
168      * <p>
169      * situation     : XAResource is associtaed with XID(1)
170      * flags=TMNOFLAGS  with XID(1)
171      * result        : XAResource creates a new transactional context
172      * </p>
173      * <p/>
174      * <p>
175      * situation     : XAResource is associtaed with XID(1)
176      * flags=TMJOIN/TNRESUME  with XID(1) (may be different in branch)
177      * result        : XAResource joins the transactional context with the XAResource already associated with XID(1)
178      * If TMRESUME is specified, start is to resume a suspended transaction branch specified in xid.
179      * </p>
180      * <p/>
181      * <p/>
182      * <p/>
183      * <p/>
184      * This method starts work on behalf of a transaction branch.
185      * <p/>
186      * <p/>
187      * <p/>
188      * If TMJOIN is specified, start is for joining an exisiting transaction branch xid.
189      * <p/>
190      * If neither TMJOIN nor TMRESUME is specified and the transaction branch specified in
191      * xid already exists, the resource manager throw the XAException
192      * with XAER_DUPID error code.
193      * <p/>
194      * If the XAResource has a current connection  and flags ==TMJOIN/TMRESUME, the current connection is
195      * substituted by the connection of the existing TX.
196      *
197      * @param xid   A global transaction identifier to be associated with the resource.
198      * @param flags One of TMNOFLAGS, TMJOIN, or TMRESUME.
199      * @throws: XAException An error has occurred.
200      * Possible exceptions are XA_RB*, XAER_RMERR, XAER_RMFAIL, XAER_DUPID, XAER_OUTSIDE, XAER_NOTA, XAER_INVAL, or XAER_PROTO.
201      */
202     public void start(Xid xid, int flags) throws XAException {
203         try {
204 
205             if (xid == null) {
206                 throw new XAException(XAException.XAER_INVAL);
207             }
208             // Find the current Branch
209             XATransactionalBranch<C> transactionalBranch = null;
210 
211             // if resuming or joining an existing transaction
212             if (flags == TMRESUME) {
213                 this.xaConnectionHandle.resumeTransactionalBranch(xid);
214             } else if (flags == TMJOIN || flags == TMNOFLAGS) {
215                 this.xaConnectionHandle.startTransactionalBranch(xid);
216             }
217             LOG.debug(
218                     "PhynixxXAResource[" + this.getId() + "]:start xid='"
219                             + xid
220                             + "' flags='"
221                             + ConstantsPrinter.getXAResourceMessage(flags)
222                             + "'"
223                             + " Connected to " +
224                             this.xaConnectionHandle);
225 
226 
227             // start monitoring the timeout
228         this.setTimeOutActive(true);
229 
230         } catch (XAException xaExc) {
231             LOG.error("PhynixxXAResource[" + this.getId() + "]:start xid='"
232                     + xid
233                     + "' flags='"
234                     + ConstantsPrinter.getXAResourceMessage(flags)
235                     + "'"
236                     + " ERROR  " + ConstantsPrinter.getXAErrorCode(xaExc.errorCode));
237             throw xaExc;
238         } catch (Exception ex) {
239             LOG.error("PhynixxXAResource.start(" + xid + "," + flags + ") on XAResourceProgressState " + this.xaId + " :: " + ex + "\n" + ExceptionUtils.getStackTrace(ex));
240             throw new DelegatedRuntimeException("start(" + xid + "," + flags + ") on XAResourceProgressState " + this.xaId, ex);
241 
242         }
243     }

246     public void commit(Xid xid, boolean onePhase) throws XAException {
247 
248         XATransactionalBranch<C> transactionalBranch = this.xaConnectionHandle.toGlobalTransactionBranch();
249         if (xid == null) {
250             LOG.error("No XID");
251             throw new XAException(XAException.XAER_INVAL);
252         }
253 
254         if (transactionalBranch == null) {
255             LOG.error("XAConnection is not associated to a global Transaction (expected to XID=" + xid + ")");
256             throw new XAException(XAException.XAER_PROTO);
257         }
258 
259         // assert that the current xaConnection is associated to this XID
260         if (!transactionalBranch.getXid().equals(xid)) {
261             LOG.error("XAResource " + this + " isnt't active for XID=" + xid);
262             throw new XAException(XAException.XAER_PROTO);
263         }
264 
265         // Find the current Branch
266 
267         try {
268             transactionalBranch.commit(onePhase);
269         } catch (XAException xaExc) {
270             LOG.error("PhynixxXAResource[" + this.getId() + "]:end xid='"
271                     + xid
272                     + "' onePhase='" + onePhase
273                     + "            ERROR  " + ConstantsPrinter.getXAErrorCode(xaExc.errorCode));
274             throw xaExc;
275 
276         } catch (Exception ex) {
277             LOG.error("PhynixxXAResource.commit(" + xid + "," + onePhase + ") on XAResourceProgressState " + this.xaId + " :: " + ex + "\n" + ExceptionUtils.getStackTrace(ex));
278             throw new DelegatedRuntimeException("commit(" + xid + "," + onePhase + ") on XAResourceProgressState " + this.xaId, ex);
279         } finally {
280             try {
281                 // Branch isn't active for the current XAresource
282                 this.xaConnectionHandle.closeTransactionalBranch(xid);
283             } finally {
284                 // stop monitoring the timeout
285             this.setTimeOutActive(false);
286 
287             }
288         }
289

292     /**
293      * finds the transactional branch of the current XAResource associated with die XID
294      * <p/>
295      * Prepares to perform a commit. May actually perform a commit
296      * in the flag commitOnPrepare is set to true.
297      * <p/>
298      * This method is called to ask the resource manager to prepare for a
299      * transaction commit of the transaction specified in xid.
300      *
301      * @param xid A global transaction identifier.
302      * @return A value indicating the resource manager's vote on the outcome of
303      * the transaction. The possible values are: XA_RDONLY or XA_OK.
304      * If the resource manager wants to roll back the transaction, it should do so by
305      * throwing an appropriate XAException in the prepare method.
306      * @throws XAException An error has occurred.
307      *                     Possible exception values are:
308      *                     XA_RB*, XAER_RMERR, XAER_RMFAIL, XAER_NOTA, XAER_INVAL, or XAER_PROTO.
309      */
310     public int prepare(Xid xid) throws XAException {
311         try {
312             LOG.debug(
313                     "PhynixxXAResource[" + this.getId() + "]:prepare prepare to perform a commit for XID=" + xid);
314 
315             XATransactionalBranch<C> transactionalBranch = this.xaConnectionHandle.toGlobalTransactionBranch();
316             if (xid == null) {
317                 LOG.error("No XID");
318                 throw new XAException(XAException.XAER_INVAL);
319             }
320 
321             if (transactionalBranch == null) {
322                 LOG.error("XAConnection is not associated to a global Transaction");
323                 throw new XAException(XAException.XAER_PROTO);
324             }
325 
326             // assert that the current xaConnection is associated to this XID
327             if (!transactionalBranch.getXid().equals(xid)) {
328                 LOG.error("XAResource " + this + " isnt't active for XID=" + xid);
329                 throw new XAException(XAException.XAER_PROTO);
330             }
331 
332 
333             // must find connection for this transaction
334             int retVal = transactionalBranch.prepare();
335 
336             if (retVal == XAResource.XA_RDONLY) {
337                 this.xaConnectionHandle.closeTransactionalBranch(xid);
338             }
339             return retVal;
340 
341         } catch (XAException xaExc) {
342             LOG.error("PhynixxXAResource[" + this.getId() + "]:prepare xid='"
343                     + xid
344                     + " ERROR  " + ConstantsPrinter.getXAErrorCode(xaExc.errorCode));
345             throw xaExc;
346         } catch (Exception ex) {
347             LOG.error("PhynixxXAResource.prepare(" + xid + ") on XAResourceProgressState " + this.xaId + " :: " + ex + "\n" + ExceptionUtils.getStackTrace(ex));
348             throw new DelegatedRuntimeException("prepare(" + xid + ") on XAResourceProgressState " + this.xaId, ex);
349 
350         }
351     }

354     /*
355     * finds the transactional branch of the current XAResource associated with die XID
356     *
357     * This method informs the resource manager to roll back
358     * work done on behalf of a transaction branch.
359     * Upon return, the resource manager has rolled back the branch's work and
360     * has released all held resources.
361     *
362     * @param xid A global transaction identifier.
363     * @throws XAException An error has occurred.
364     *    Possible XAExceptions are
365     *       XA_HEURHAZ, XA_HEURCOM,XA_HEURRB, XA_HEURMIX,
366     *       XAER_RMERR, XAER_RMFAIL, XAER_NOTA,XAER_INVAL, or XAER_PROTO.
367     *
368     *
369     * @see javax.transaction.xa.XAResource#rollback(javax.transaction.xa.Xid)
370     */
371     public void rollback(Xid xid) throws XAException {
372         {
373             try {
374                 LOG.debug("PhynixxXAResource[" + this.getId() + "]:rollback started xid=" + xid);
375 
376                 XATransactionalBranch<C> transactionalBranch = this.xaConnectionHandle.toGlobalTransactionBranch();
377                 if (xid == null) {
378                     LOG.error("No XID");
379                     throw new XAException(XAException.XAER_INVAL);
380                 }
381 
382                 if (transactionalBranch == null) {
383                     LOG.error("XAConnection is not associated to a global Transaction");
384                     throw new XAException(XAException.XAER_PROTO);
385                 }
386 
387                 // assert that the current xaConnection is associated to this XID
388                 if (!transactionalBranch.getXid().equals(xid)) {
389                     LOG.error("XAResource " + this + " isnt't active for XID=" + xid);
390                     throw new XAException(XAException.XAER_PROTO);
391                 }
392 
393                 // Find the current Branch
394 
395                 if (transactionalBranch.isInActive()) {
396                     throw new XAException(XAException.XAER_PROTO);
397                 }
398 
399                 try {
400                     transactionalBranch.rollback();
401                 } finally {
402                     // Branch isn't active for the current XAResource
403                     this.xaConnectionHandle.closeTransactionalBranch(xid);
404                 }
405 
406 
407             } catch (XAException xaExc) {
408                 LOG.error("PhynixxXAResource[" + this.getId() + "]:rollback xid='"
409                         + xid
410                         + " ERROR  " + ConstantsPrinter.getXAErrorCode(xaExc.errorCode));
411                 throw xaExc;
412 
413             } catch (Exception ex) {
414                 LOG.error("PhynixxXAResource.rollback(" + xid + ") on XAResourceProgressState " + this.xaId + " :: " + ex + "\n" + ExceptionUtils.getStackTrace(ex));
415                 throw new DelegatedRuntimeException("rollback(" + xid + ") on XAResourceProgressState " + this.xaId, ex);
416 
417             } finally {
418                 // stop monitoring the timeout
419             this.setTimeOutActive(false);
420 
421             }
422         }

427     /**
428      * This method ends the work performed on behalf of a transaction branch.
429      * <p/>
430      * The resource manager dissociates the XA resource from the transaction branch specified and let the transaction
431      * be completed.
432      * If TMSUSPEND is specified in flags, the transaction branch is temporarily suspended in incomplete state.
433      * The transaction context is in suspended state and must be resumed via start with TMRESUME specified.
434      * <p/>
435      * If TMFAIL is specified, the portion of work has failed. The resource manager
436      * may mark the transaction as rollback only.
437      * <p/>
438      * <p> the spec doesn't no fix the order of commit/rollback and end and we have to take of both orders.
439      * If end comes before rollback/commit, the transactional branch has to be freeze.
440      * If rollback/commit comes before end, the transactional branch has to be closed
441      * </p>
442      * If TMSUCCESS is specified, the portion of work has completed successfully. end is called in Transaction.delistResource
443      *
444      * @param xid   A global transaction identifier.
445      * @param flags If true, the resource manager should use a one-phase commit protocol
446      *              to commit the work done on behalf of xid.
447      * @throws: XAException An error has occurred. Possible XAException values
448      * are XAER_RMERR, XAER_RMFAIL,XAER_NOTA, XAER_INVAL, XAER_PROTO,XA_RB*.
449      */
450     public void end(Xid xid, int flags) throws XAException {
451         try {
452             //not tested XS
453             LOG.debug("PhynixxXAResource:end");
454             LOG.debug(
455                     "PhynixxXAResource[" + this.getId() + "]:end xid='" + xid + "' flags='" + ConstantsPrinter.getXAResourceMessage(flags) + "'");
456 
457             if (xid == null) {
458                 LOG.error("No XID");
459                 throw new XAException(XAException.XAER_INVAL);
460             }
461 
462             XATransactionalBranch<C> transactionalBranch = this.xaConnectionHandle.toGlobalTransactionBranch();
463             if (transactionalBranch == null) {
464                 LOG.error("XAConnection is not associated to a global Transaction");
465                 throw new XAException(XAException.XAER_PROTO);
466             }
467 
468             // assert that the current xaConnection is associated to this XID
469             if (!transactionalBranch.getXid().equals(xid)) {
470                 LOG.error("XAResource " + this + " isnt't active for XID=" + xid);
471                 throw new XAException(XAException.XAER_PROTO);
472             }
473 
474             if (flags == TMSUSPEND) {
475                 this.xaConnectionHandle.suspendTransactionalBranch(xid);
476             } else if (flags == TMSUCCESS) {
477 
478                 // XAProtocol is finsihed an the branch isnt needed any longer
479                 if (transactionalBranch.isXAProtocolFinished()) {
480                     transactionalBranch.close();
481                 }
482                 LOG.error("XAResource " + this + " closed gracefully ");
483             } else if (flags == TMFAIL) {
484                 transactionalBranch.setRollbackOnly(true);
485             }
486 
487         } catch (XAException xaExc) {
488             LOG.error("PhynixxXAResource[" + this.getId() + "]:end xid='"
489                     + xid
490                     + "' flags='"
491                     + ConstantsPrinter.getXAResourceMessage(flags)
492                     + "'"
493                     + " ERROR  " + ConstantsPrinter.getXAErrorCode(xaExc.errorCode));
494             throw xaExc;
495         } catch (Exception ex) {
496             LOG.error("PhynixxXAResource.end(" + xid + "," + flags + ") on XAResourceProgressState " + this.xaId + " :: " + ex + "\n" + ExceptionUtils.getStackTrace(ex));
497             throw new DelegatedRuntimeException("end(" + xid + "," + flags + ") on XAResourceProgressState " + this.xaId, ex);
498         }
499     }

531     /**
532      * This method is called to determine if the resource
533      * manager instance represented by the target object is the same
534      * as the resource manager instance represented by the parameter xares .
535      * <p/>
536      * The resource manager is reresented by the ResourceFactory.
537      *
538      * @param xaResource An XAResource object.
539      * @return true if same RM instance; otherwise false.
540      */
541 
542     public boolean isSameRM(XAResource xaResource) throws XAException {
543         if (this.equals(xaResource)) { // if the same object
544             LOG.debug("PhynixxXAResource[" + this.getId() + "]:isSameRM isSameRM");
545             return true; // then definitely the same RM
546         }
547         if (!(xaResource instanceof PhynixxXAResource)) {
548             // if it's not one of our wrappers
549             LOG.debug("PhynixxXAResource[" + this.getId() + "]:isSameRM not isSameRM");
550             return false; // then it's definitely not the same RM
551         }
552         PhynixxXAResource sampleXARes = (PhynixxXAResource) xaResource;
553 
554         try {
555             // cast to something more convenient
556             if (xaResourceFactory.equals(sampleXARes.xaResourceFactory)) {
557                 // if they originate from same data source
558                 LOG.debug("PhynixxXAResource[" + this.getId() + "]:isSameRM isSameRM (equal XAResourceFactory)");
559                 return true; // then they're the same RM
560             } else {
561                 LOG.debug("PhynixxXAResource[" + this.getId() + "]:isSameRM not isSameRM (not equal XAResourceFactory)");
562                 return false;
563             }
564         } catch (Exception ex) {
565             LOG.error("PhynixxXAResource.isSameRM(" + sampleXARes.xaId + ") on XAResourceProgressState " + this.xaId + " :: " + ex + "\n" + ExceptionUtils.getStackTrace(ex));
566             throw new DelegatedRuntimeException("isSameRM(" + sampleXARes.xaId + ") on XAResourceProgressState " + this.xaId, ex);
567 
568         }
569     }

592     /**
593      * the system is recovered by the xaResourceFactory representing the persistence management system,
594      */
595     public Xid[] recover(int flags) throws XAException {
596         LOG.info("PhynixxXAResource[" + this.getId() + "]:recover recover flags=" + ConstantsPrinter.getXAResourceMessage(flags));
597         if (flags != TMSTARTRSCAN && flags != TMENDRSCAN && flags != TMNOFLAGS) {
598             throw new XAException(XAException.XAER_INVAL);
599         }
