54      protected void renew(int count) {
55          this.releaseAll();
56          this.testers = new WatchdogAware[count];
57          for (int i = 0; i < count; i++) {
58              testers[i] = this.startWatchDog();
59          }

67      private void releaseAll() {
68          for (int i = 0; i < testers.length; i++) {
69              testers[i].kill();
70              WatchdogRegistry.getTheRegistry().shutdown(testers[i].getWatchdog().getId());
71          }
72  
73          // release all references to release weakly references Watchdogs
74          testers = new WatchdogAware[]{};

89      private static void gc() {
90          Long l = new Long(3);
91          l = null;
92          List byteArrays = new ArrayList();
93          SoftReference sref = new SoftReference(new char[]{'1'});
94          System.out.print("GC initialized ");
95          while (sref.get() != null) {
96              //System.out.println("Available Heap Size= "+Runtime.getRuntime().freeMemory()+" of "+Runtime.getRuntime().totalMemory());
97              byte[] b = new byte[1024 * 1024];
98              byteArrays.add(b);
99              System.out.print(".");
100             System.gc();
101             try {
102                 Thread.currentThread().sleep(10);
103             } catch (InterruptedException e) {
104             }
105         }
106 
107         System.out.print("finished");
108     }

156     public void control() throws Exception {
157 
158         System.out.println(usage());
159         while (true) {
160             try {
161                 String cmd = this.readIn();
162                 if (cmd.equals("u")) {
163                     System.out.println(usage());
164                 } else if (cmd.equals("q")) {
165                     this.printWatchdogInfos();
166                     System.exit(1);
167                 } else if (cmd.equals("rl")) {
168                     this.releaseAll();
169                 } else if (cmd.equals("r")) {
170                     WatchdogRegistry.getTheRegistry().restart();
171                 } else if (cmd.equals("sh")) {
172                     WatchdogRegistry.getTheRegistry().shutdown();
173                 } else if (cmd.equals("st")) {
174                     WatchdogRegistry.getTheRegistry().stop();
175                 } else if (cmd.equals("i")) {
176                     this.printWatchdogInfos();
177                 } else if (cmd.equals("gc")) {
178                     WatchdogConsole.gc();
179                 } else if (cmd.equals("mg")) {
180                     printWatchdogManagementInfo();
181                 } else if (cmd.startsWith("sh")) {
182                     WatchdogRegistry.getTheRegistry().shutdown(extractNumber(cmd));
183                 } else if (cmd.startsWith("r")) {
184                     WatchdogRegistry.getTheRegistry().restart(extractNumber(cmd));
185                 } else if (cmd.startsWith("st")) {
186                     WatchdogRegistry.getTheRegistry().stop(extractNumber(cmd));
187                 } else if (cmd.startsWith("i")) {
188                     this.printWatchdogInfos(extractNumber(cmd));
189                 } else if (cmd.startsWith("n")) {
190                     this.renew(extractNumber(cmd).intValue());
191                 } else if (cmd.startsWith("mg")) {
192                     WatchdogRegistry.setWatchdogManagementInterval(extractNumber(cmd).longValue());
193                 } else {
194                     System.out.println("Command " + cmd + " is not recognized");
195                 }
196             } catch (Throwable th) {
197                 th.printStackTrace();
198             }
199 
200         }
201     }

203     private Long extractNumber(String cmd) {
204         int i = 0;
205         for (i = 0; i < cmd.length(); i++) {
206             if (Character.isDigit(cmd.charAt(i))) {
207                 break;
208             }
209         }
210         return new Long(Long.parseLong(cmd.substring(i, cmd.length())));
211     }

213     private String readIn() throws IOException {
214 
215         char c = (char) System.in.read();
216         StringBuffer buffer = new StringBuffer();
217         buffer.append(c);
218 
219         while (System.in.available() > 0) {
220             c = (char) System.in.read();
221             if (Character.isLetter(c) || Character.isDigit(c)) {
222                 buffer.append(c);
223             }
224         }
225         // System.out.println("Read "+ buffer.toString());
226         return buffer.toString();
227     }
