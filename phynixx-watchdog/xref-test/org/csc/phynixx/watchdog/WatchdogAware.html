28  import org.csc.phynixx.common.generator.IDGenerators;
29  import org.csc.phynixx.common.logger.IPhynixxLogger;

33  /**
34   * this class starts both a thread and a watchdog watching itsself.
35   * It can must be explicitly killed.
36   * The watchdog is killed, if the C'tor parameter removeConditionOnExit is true.
37   * <p/>
38   * If not the registry waits to the condition to be finalized and the condition to become useless.
39   *
40   * @author christoph
41   */
42  public class WatchdogAware extends TimeoutCondition implements Runnable, IWatchedCondition {
43  
44      private static final IDGenerator<Long> ID_GENERATOR = IDGenerators.createLongGenerator(1, true);
45  
46      private static long CHECK_INTERVAL = 10;
47  
48      private IPhynixxLogger log = PhynixxLogManager.getLogger(this.getClass());
49      private long idleTime = -1l;
50  
51      private volatile boolean killed = false;
52  
53      private volatile byte conditionFailedCounter = 0;
54      private volatile boolean shutdown = false;
55  
56      private IWatchdog watchdog = null;
57  
58      private String id = null;
59  
60  
61      /**
62       * @param idleTime idle time of this runnable
63       * @param timeout  timeout of the wachdog condition
64       */
65      public WatchdogAware(long idleTime, long timeout) {
66          super(timeout);
67          this.idleTime = idleTime;
68          this.id = "WatchdogAware[" + ID_GENERATOR.generate() + "]";
69      }
70  
71      public void kill() {
72          this.killed = true;
73      }
74  
75  
76      public boolean isKilled() {
77          return this.killed;
78      }
79  
80      public long getIdleTime() {
81          return idleTime;
82      }
83  
84  
85      public IWatchdog getWatchdog() {
86          return watchdog;
87      }
88  
89  
90      public synchronized int getConditionFailedCounter() {
91          return new Byte(this.conditionFailedCounter).intValue();
92      }
93  
94      public synchronized boolean isShutdown() {
95          return shutdown;
96      }
97  
98      /**
99       * registers an Condition just counting the calls to check the it
100      *
101      */
102     private void registerWatching() {
103         IWatchdog wd = WatchdogRegistry.getTheRegistry().createWatchdog(CHECK_INTERVAL);
104         this.setActive(true);
105         wd.registerCondition(this);
106         this.watchdog = wd;
107     }
108 
109     public void run() {
110         this.killed = false;
111 
112         this.registerWatching();
113 
114 
115         log.info(this.id + " started (idletime=" + idleTime + " checkInterval=" + super.getTimeout() + ")");
116 
117         // System.out.println("Thread "+Thread.currentThread()+" is Killed "+ killed);
118         while (!this.isKilled()) {
119             // wait until it time to check the conditions
120             long start = System.currentTimeMillis();
121             long waiting = this.idleTime;
122 
123             while (waiting > 0) {
124                 try {
125                     Thread.sleep(waiting);
126                 } catch (InterruptedException e) {
127                 } finally {
128                     waiting = this.idleTime - (System.currentTimeMillis() - start);
129                 }
130             }
131 
132             // log.debug("WatchdogAware "+id+" - Thread " + Thread.currentThread()+" -- live time "+ System.currentTimeMillis()+" interval="+(System.currentTimeMillis() - start) );
133         }
134         log.info(this.id + " is killed");
135         this.shutdown = true;
136 
137     }
138 
139     protected void finalize() throws Throwable {
140         log.error(this.id + " is finalized");
141         super.finalize();
142     }
143 
144     public void conditionViolated() {
145         this.resetCondition();
146         this.conditionFailedCounter++;
147         log.info(new ConditionViolatedLog(this, this.id));
148     }
